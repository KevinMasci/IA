% regla para buscar electrodomesticos segun sus caracteristicas
buscar_electrodomestico(Tipo, CapacidadMin, Eficiencia, FuncionesDeseadas, PrecioMax, Inteligente, Resultado) :-
    electrodomestico(Resultado, Tipo, Capacidad, EficienciaEnergetica, Funciones, Precio, _, Inteligente),
    Capacidad >= CapacidadMin,
    Precio =< PrecioMax,
    eficiencia_compatible(EficienciaEnergetica, Eficiencia),
    funciones_compatibles(Funciones, FuncionesDeseadas).

% verifica que la eficiencia energetica sea igual o superior
eficiencia_compatible(Eficiencia, EficienciaRequerida) :-
    ordenar_eficiencia(Eficiencia, NivelActual),
    ordenar_eficiencia(EficienciaRequerida, NivelRequerido),
    NivelActual =< NivelRequerido.

% niveles de eficiencia energetica en orden
ordenar_eficiencia('a+++ ', 1).
ordenar_eficiencia('a++', 2).
ordenar_eficiencia('a+', 3).
ordenar_eficiencia('a', 4).
ordenar_eficiencia('b', 5).

% verifica si las funciones solicitadas estan en la lista de funciones del electrodomestico
funciones_compatibles([], _).
funciones_compatibles([Funcion|T], FuncionesDeseadas) :-
    member(Funcion, FuncionesDeseadas),
    funciones_compatibles(T, FuncionesDeseadas).
